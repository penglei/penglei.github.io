data Document = Document
  { docExt :: Maybe String,
    docTitle :: Text,
    docAuthor :: Maybe Text,
    docTags :: Maybe [Text],
    docDate :: Maybe Text,
    docContent :: Maybe Text,
    docPath :: Maybe Text,
    docAttrs :: Maybe Value
  }
  deriving (Show, Generic)
  deriving (FromJSON, ToJSON) via PrefixedSnake "doc" Document

qsort []     = []
qsort (x:xs) = qsort (filter (< x) xs) ++ [x] ++
               qsort (filter (>= x) xs)
